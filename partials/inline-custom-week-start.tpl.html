Example 5: Custom Week Start

HTML
<body ng-app="mainApp">
    <div ng-controller="InlineCustomWeekStartCtrl as vm">
        <input type="text"
               ng-model="vm.date"   
               angular-datepicker-light="vm.datepickerOptions" />
    </div>
</body>
JavaScript
angular
    .module('mainApp', ['datepickerLightModule'])
    .controller('InlineCustomWeekStartCtrl', InlineCustomWeekStartCtrl);
    
    function InlineCustomWeekStartCtrl() {
        var vm = this;
        
        vm.date = new Date();
        vm.datepickerOptions = {
            inline: true,
            firstDayOfWeek: 1
        };
    }
