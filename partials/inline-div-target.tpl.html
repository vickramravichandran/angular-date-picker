Using <div> Target

HTML
<body ng-app="mainApp">
    <div ng-controller="InlineDivTargetCtrl as vm">
        <div class="date-target" 
             angular-datepicker-light="vm.datepickerOptions">
        </div>
    </div>
</body>
JavaScript
angular
    .module('mainApp', ['datepickerLightModule'])
    .controller('InlineDivTargetCtrl', InlineDivTargetCtrl);
    
    function InlineDivTargetCtrl() {
        var vm = this;
        
        vm.datepickerOptions = {
            inline: true,
            dateFormat: 'MMM Do YYYY',
            containerCssClass: "datepicker-container-inline"
        };
