Example 4: Dates from Other Months

HTML
<body ng-app="mainApp">
    <div ng-controller="InlineOtherMonthDatesCtrl as vm">
        <input type="text"
               ng-model="vm.date"   
               angular-datepicker-light="vm.datepickerOptions" />
    </div>
</body>
JavaScript
angular
    .module('mainApp', ['datepickerLightModule'])
    .controller('InlineOtherMonthDatesCtrl', InlineOtherMonthDatesCtrl);
    
    function InlineOtherMonthDatesCtrl() {
        var vm = this;
        
        vm.date = new Date();
        vm.datepickerOptions = {
            inline: true,
            showOtherMonthDates: true
        };        
    }
